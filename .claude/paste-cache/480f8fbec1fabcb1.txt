 CLAUDE CODE — Générateur E-E-A-T Complet

## OBJECTIF

Créer un module **"E-E-A-T Builder"** pour CitedBy.ai qui génère tous les éléments d'autorité nécessaires pour être cité par les IA.

**E-E-A-T = Experience, Expertise, Authoritativeness, Trustworthiness**

---

## POURQUOI C'EST CRUCIAL POUR L'AEO

Les IA (ChatGPT, Claude, Perplexity) priorisent les sources avec :
- **Auteurs identifiés** avec credentials vérifiables
- **Entités reconnues** dans leur domaine
- **Preuves sociales** (avis, certifications, presse)
- **Signaux de confiance** (HTTPS, mentions légales, ancienneté)

---

## 6 OUTILS À CRÉER

### 1. Author Bio Generator
Génère une bio auteur optimisée pour chaque expert/rédacteur.

**Output :**
- Bio courte (50 mots) pour signature articles
- Bio longue (150 mots) pour page auteur
- Schema.org Person complet
- Liens sociaux structurés

### 2. About Page Generator
Génère une page "À Propos" complète avec tous les signaux E-E-A-T.

**Output :**
- HTML complet de la page
- Schema.org Organization/Person
- Blocs : Histoire, Équipe, Mission, Chiffres clés, Certifications

### 3. Trust Signals Generator
Génère les blocs de confiance à intégrer sur le site.

**Output :**
- Bloc "Certifications & Partenaires"
- Bloc "Ils parlent de nous" (presse)
- Bloc "Avis clients" avec Schema.org
- Badges de confiance (garanties, paiement sécurisé)

### 4. Expert Credentials Generator
Génère les credentials détaillées pour chaque expert.

**Output :**
- Liste des diplômes (Schema.org `hasCredential`)
- Formations (`alumniOf`)
- Récompenses (`award`)
- Publications (`author` de `ScholarlyArticle`)

### 5. Source Citation Block Generator
Génère le bloc "À propos de l'auteur" pour les articles.

**Output :**
- Encart auteur pour fin d'article
- Mini-bio avec photo
- Liens vers page auteur et LinkedIn
- Date de dernière vérification

### 6. E-E-A-T Audit & Score
Analyse une URL et génère un score E-E-A-T avec recommandations.

**Output :**
- Score global /100
- Scores par pilier (Experience, Expertise, Authority, Trust)
- Liste des éléments manquants
- Actions prioritaires

```typescript
// types/eeat.ts

// === AUTEUR / PERSONNE ===
interface AuthorProfile {
  id: string;
  name: string;
  slug: string;
  jobTitle: string;
  organization: string;
  
  // Experience
  yearsExperience: number;
  specializations: string[];
  notableProjects?: string[];
  
  // Expertise
  credentials: Credential[];
  certifications: Certification[];
  education: Education[];
  publications?: Publication[];
  speakingEngagements?: Event[];
  
  // Authority
  socialProfiles: SocialProfile[];
  mediaAppearances?: MediaMention[];
  awards?: Award[];
  
  // Trust
  verifiedBy?: string;
  lastVerified?: string;
  
  // Media
  photoUrl?: string;
  
  // Bios générées
  bioShort?: string;  // 50 mots
  bioLong?: string;   // 150 mots
}

interface Credential {
  type: 'degree' | 'certification' | 'license' | 'membership';
  name: string;
  issuer: string;
  year?: number;
  url?: string;
  credentialId?: string;
}

interface Education {
  degree: string;
  field: string;
  institution: string;
  year: number;
  honors?: string;
}

interface Certification {
  name: string;
  issuer: string;
  dateObtained: string;
  dateExpires?: string;
  credentialUrl?: string;
}

interface Publication {
  title: string;
  type: 'book' | 'article' | 'paper' | 'study';
  publisher?: string;
  date: string;
  url?: string;
  coAuthors?: string[];
}

interface Award {
  name: string;
  issuer: string;
  year: number;
  description?: string;
}

interface SocialProfile {
  platform: 'linkedin' | 'twitter' | 'github' | 'youtube' | 'other';
  url: string;
  followers?: number;
}

interface MediaMention {
  outlet: string;
  title: string;
  date: string;
  url?: string;
  type: 'interview' | 'quote' | 'feature' | 'podcast';
}

// === ORGANISATION ===
interface OrganizationProfile {
  name: string;
  legalName?: string;
  type: 'Organization' | 'LocalBusiness' | 'Corporation';
  
  // Experience
  foundingDate: string;
  industry: string;
  employeeCount?: { min: number; max: number };
  
  // Expertise
  specializations: string[];
  servicesOffered: string[];
  
  // Authority
  certifications: Certification[];
  awards: Award[];
  partnerships: string[];
  mediaAppearances: MediaMention[];
  clients?: string[]; // Logos clients notables
  
  // Trust
  reviews: ReviewAggregate[];
  guarantees: string[];
  accreditations: string[];
  
  // Contact
  address: Address;
  phone?: string;
  email: string;
  
  // Social
  socialProfiles: SocialProfile[];
  
  // Media
  logoUrl?: string;
}

interface ReviewAggregate {
  platform: 'google' | 'trustpilot' | 'facebook' | 'yelp' | 'other';
  rating: number;
  count: number;
  url?: string;
}

interface Address {
  street: string;
  city: string;
  postalCode: string;
  country: string;
}

// === TRUST SIGNALS ===
interface TrustSignals {
  certifications: {
    name: string;
    logo?: string;
    issuer: string;
    verifyUrl?: string;
  }[];
  
  securityBadges: {
    type: 'ssl' | 'pci' | 'gdpr' | 'secure_payment';
    provider?: string;
  }[];
  
  guarantees: {
    type: 'money_back' | 'satisfaction' | 'warranty' | 'free_returns';
    duration?: string;
    details: string;
  }[];
  
  socialProof: {
    metric: string; // "50,000+ clients"
    icon?: string;
  }[];
  
  pressMentions: {
    outlet: string;
    logo?: string;
    quote?: string;
    url?: string;
  }[];
}

// === E-E-A-T AUDIT ===
interface EeatAudit {
  url: string;
  date: string;
  
  scores: {
    overall: number;
    experience: number;
    expertise: number;
    authoritativeness: number;
    trustworthiness: number;
  };
  
  findings: {
    category: 'experience' | 'expertise' | 'authority' | 'trust';
    status: 'present' | 'missing' | 'incomplete';
    element: string;
    impact: 'high' | 'medium' | 'low';
    recommendation?: string;
  }[];
  
  priorityActions: {
    action: string;
    impact: number;
    effort: 'low' | 'medium' | 'high';
    tool: string; // Quel outil EEAT utiliser
  }[];
}

// === OUTPUTS GÉNÉRÉS ===
interface GeneratedAuthorBio {
  bioShort: string;
  bioLong: string;
  schemaOrg: string; // JSON-LD Person
  htmlBlock: string; // Bloc HTML prêt à intégrer
}

interface GeneratedAboutPage {
  html: string;
  schemaOrg: string;
  sections: {
    hero: string;
    story: string;
    team: string;
    values: string;
    numbers: string;
    certifications: string;
    cta: string;
  };
}

interface GeneratedTrustBlock {
  html: string;
  css: string;
  schemaOrg?: string;
}

interface GeneratedCitationBlock {
  html: string;
  schemaOrg: string;
}
```

## OBJECTIF

Créer un module **"E-E-A-T Builder"** pour CitedBy.ai qui génère tous les éléments d'autorité nécessaires pour être cité par les IA.

**E-E-A-T = Experience, Expertise, Authoritativeness, Trustworthiness**

---

## POURQUOI C'EST CRUCIAL POUR L'AEO

Les IA (ChatGPT, Claude, Perplexity) priorisent les sources avec :
- **Auteurs identifiés** avec credentials vérifiables
- **Entités reconnues** dans leur domaine
- **Preuves sociales** (avis, certifications, presse)
- **Signaux de confiance** (HTTPS, mentions légales, ancienneté)

---

## 6 OUTILS À CRÉER

### 1. Author Bio Generator
Génère une bio auteur optimisée pour chaque expert/rédacteur.

**Output :**
- Bio courte (50 mots) pour signature articles
- Bio longue (150 mots) pour page auteur
- Schema.org Person complet
- Liens sociaux structurés

### 2. About Page Generator
Génère une page "À Propos" complète avec tous les signaux E-E-A-T.

**Output :**
- HTML complet de la page
- Schema.org Organization/Person
- Blocs : Histoire, Équipe, Mission, Chiffres clés, Certifications

### 3. Trust Signals Generator
Génère les blocs de confiance à intégrer sur le site.

**Output :**
- Bloc "Certifications & Partenaires"
- Bloc "Ils parlent de nous" (presse)
- Bloc "Avis clients" avec Schema.org
- Badges de confiance (garanties, paiement sécurisé)

### 4. Expert Credentials Generator
Génère les credentials détaillées pour chaque expert.

**Output :**
- Liste des diplômes (Schema.org `hasCredential`)
- Formations (`alumniOf`)
- Récompenses (`award`)
- Publications (`author` de `ScholarlyArticle`)

### 5. Source Citation Block Generator
Génère le bloc "À propos de l'auteur" pour les articles.

**Output :**
- Encart auteur pour fin d'article
- Mini-bio avec photo
- Liens vers page auteur et LinkedIn
- Date de dernière vérification

### 6. E-E-A-T Audit & Score
Analyse une URL et génère un score E-E-A-T avec recommandations.

**Output :**
- Score global /100
- Scores par pilier (Experience, Expertise, Authority, Trust)
- Liste des éléments manquants
- Actions prioritaires

```typescript
  type: 'book' | 'article' | 'paper' | 'study';
  publisher?: string;
  date: string;
  url?: string;
  coAuthors?: string[];
}

interface Award {
  name: string;
  issuer: string;
  year: number;
  description?: string;

  followers?: number;

  date: string;
  url?: string;

}

  legalName?: string;
  type: 'Organization' | 'LocalBusiness' | 'Corporation';

  foundingDate: string;
  industry: string;
  employeeCount?: { min: number; max: number };
  
  // Expertise
  specializations: string[];
  servicesOffered: string[];
  
  // Authority
  certifications: Certification[];
  awards: Award[];
  partnerships: string[];
  mediaAppearances: MediaMention[];
  clients?: string[]; // Logos clients notables
  
  // Trust
  reviews: ReviewAggregate[];
  guarantees: string[];
  accreditations: string[];
  
  // Contact
  address: Address;
  phone?: string;
  email: string;
  
  // Social
  socialProfiles: SocialProfile[];
  
  // Media
  logoUrl?: string;
}

interface ReviewAggregate {
  platform: 'google' | 'trustpilot' | 'facebook' | 'yelp' | 'other';
  rating: number;
  count: number;
  url?: string;
}

  street: string;
  city: string;
  postalCode: string;
  country: string;
}

  certifications: {
    name: string;
    logo?: string;
    issuer: string;
    verifyUrl?: string;
  }[];
  
  securityBadges: {
    type: 'ssl' | 'pci' | 'gdpr' | 'secure_payment';
    provider?: string;
  }[];
  
  guarantees: {
    type: 'money_back' | 'satisfaction' | 'warranty' | 'free_returns';
    duration?: string;
    details: string;
  }[];
  
  socialProof: {
    metric: string; // "50,000+ clients"
    icon?: string;
  }[];
  
  pressMentions: {
    outlet: string;
    logo?: string;
    quote?: string;
    url?: string;
  }[];
}

interface EeatAudit {
  url: string;
  date: string;
  
  scores: {
    overall: number;
    experience: number;
    expertise: number;
    authoritativeness: number;
    trustworthiness: number;
  };
  
  findings: {
    category: 'experience' | 'expertise' | 'authority' | 'trust';
    status: 'present' | 'missing' | 'incomplete';
    element: string;
    impact: 'high' | 'medium' | 'low';
    recommendation?: string;
  }[];
  
  priorityActions: {
    action: string;
    impact: number;
    effort: 'low' | 'medium' | 'high';
    tool: string; // Quel outil EEAT utiliser
  }[];
}

}
```

```typescript
  partnerships: string[];
  mediaAppearances: MediaMention[];
  clients?: string[]; // Logos clients notables
  
  // Trust
  reviews: ReviewAggregate[];
  guarantees: string[];
  accreditations: string[];
  
  // Contact
  address: Address;
  phone?: string;
  email: string;
  
  // Social
  socialProfiles: SocialProfile[];
  
  // Media
  logoUrl?: string;
}

interface ReviewAggregate {
  platform: 'google' | 'trustpilot' | 'facebook' | 'yelp' | 'other';
  rating: number;
  count: number;
  url?: string;
}

interface Address {
  street: string;
  city: string;
  postalCode: string;
  country: string;
}

// === TRUST SIGNALS ===
interface TrustSignals {
  certifications: {
    name: string;
    logo?: string;
    issuer: string;
    verifyUrl?: string;
  }[];
  
  securityBadges: {
    type: 'ssl' | 'pci' | 'gdpr' | 'secure_payment';
    provider?: string;
  }[];
  
  guarantees: {
    type: 'money_back' | 'satisfaction' | 'warranty' | 'free_returns';
    duration?: string;
    details: string;
  }[];
  
  socialProof: {
    metric: string; // "50,000+ clients"
    icon?: string;
  }[];
  
  pressMentions: {
    outlet: string;
    logo?: string;
    quote?: string;
    url?: string;
  }[];
}

// === E-E-A-T AUDIT ===
interface EeatAudit {
  url: string;
  date: string;
  
  scores: {
    overall: number;
    experience: number;
    expertise: number;
    authoritativeness: number;
    trustworthiness: number;
  };
  
  findings: {
    category: 'experience' | 'expertise' | 'authority' | 'trust';
    status: 'present' | 'missing' | 'incomplete';
    element: string;
    impact: 'high' | 'medium' | 'low';
    recommendation?: string;
  }[];
  
  priorityActions: {
    action: string;
    impact: number;
    effort: 'low' | 'medium' | 'high';
    tool: string; // Quel outil EEAT utiliser
  }[];
}

  };
}
```
