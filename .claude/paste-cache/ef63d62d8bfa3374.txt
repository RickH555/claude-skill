```typescript
// services/mega-generator.ts

// MÉGA-GÉNÉRATEUR : Un appel API = toutes les solutions pré-générées

// ═══════════════════════════════════════════════

export interface MegaGenerationResult {

  generatedAt: string;
  url: string;
  apiModel: string;
  tokenCount: number;
  
  // Solutions pré-générées (toutes verrouillées par défaut)

  // Contenus complets pré-générés

  };

  };

// ═══════════════════════════════════════════════

const MEGA_SYSTEM_PROMPT = `Tu es un expert en AEO (Answer Engine Optimization) et en optimisation de contenu pour les moteurs de réponse IA (ChatGPT, Perplexity, Gemini).

const getApiKey = (): string => {
  const key = import.meta.env.VITE_GEMINI_API_KEY;
  if (!key) throw new Error('VITE_GEMINI_API_KEY non configurée');
  return key;
};

/**
 * Génère TOUTES les solutions en UN SEUL appel API.
 * 
 * @param url - URL analysée
 * @param pageTitle - Titre de la page détecté par l'analyseur
 * @param pageContent - Contenu textuel extrait (sera tronqué à 3000 chars)
 * @param failedCriteria - Liste des critères échoués issus de l'audit
 * @param detectedProducts - Produits/services détectés sur la page
 * @param detectedBrand - Nom de marque détecté
 * @returns Toutes les solutions pré-générées
 */
export async function generateAllSolutions(
  url: string,
  pageTitle: string,
  pageContent: string,
  failedCriteria: Array<{ id: string; name: string; category: string; maxPoints: number }>,
  detectedProducts: string[] = [],
  detectedBrand: string = '',
  language: string = 'fr'
): Promise<MegaGenerationResult> {

  const apiKey = getApiKey();
  
  const prompt = buildMegaPrompt(
    url, pageTitle, pageContent,
    failedCriteria, detectedProducts, detectedBrand, language
  );

  try {
    const ai = new GoogleGenAI({ apiKey });
    
    const response = await ai.models.generateContent({
      model: 'gemini-2.0-flash',
      contents: prompt,
      config: {
        systemInstruction: MEGA_SYSTEM_PROMPT,
        maxOutputTokens: 16000,  // On veut une réponse longue et complète
        temperature: 0.3,         // Basse température = plus factuel et structuré
      },
    });

    const text = response.text || '';
    
    // Nettoyer le JSON (supprimer backticks markdown si présents)
    const cleanJson = text
      .replace(/```json\n?/g, '')
      .replace(/```\n?/g, '')
      .trim();
    
    const parsed = JSON.parse(cleanJson);

    // Valider et enrichir le résultat
    const result: MegaGenerationResult = {
      generatedAt: new Date().toISOString(),
      url,
      apiModel: 'gemini-2.0-flash',
      tokenCount: text.length, // approximation
      
      solutions: (parsed.solutions || []).map((sol: any, index: number) => ({
        id: sol.id || `sol_${String(index + 1).padStart(3, '0')}`,
        category: sol.category || 'content',
        title: sol.title || `Solution ${index + 1}`,
        description: sol.description || '',
        priority: sol.priority || 'medium',
        estimatedImpact: sol.estimatedImpact || 2,
        fixesCriteria: sol.fixesCriteria || [],
        mostImpactedEngines: sol.mostImpactedEngines || [],
        generatedContent: {
          code: sol.generatedContent?.code || '',
          language: sol.generatedContent?.language || 'html',
          explanation: sol.generatedContent?.explanation || '',
          implementationSteps: sol.generatedContent?.implementationSteps || [],
        },
      })),
      
      fullFAQ: parsed.fullFAQ || { html: '', schemaJsonLd: '', questionsCount: 0 },
      fullSchemaOrg: parsed.fullSchemaOrg || { jsonLd: '', types: [] },
      fullMetaTags: parsed.fullMetaTags || { title: '', description: '', openGraph: '', canonical: '' },
      fullIntroduction: parsed.fullIntroduction || { html: '', wordCount: 0 },
      fullComparativeTable: parsed.fullComparativeTable || { html: '', columnsCount: 0, rowsCount: 0 },
      fullArticleOutline: parsed.fullArticleOutline || { markdown: '', sections: [], estimatedWordCount: 0 },
      
      solutionsCount: (parsed.solutions || []).length,
      totalEstimatedImpact: (parsed.solutions || []).reduce(
        (sum: number, s: any) => sum + (s.estimatedImpact || 0), 0
      ),
    };

    return result;
    
  } catch (error) {
    console.error('Erreur méga-génération:', error);
    // Fallback : retourner un résultat vide mais valide
    return generateFallbackResult(url, failedCriteria);
  }
}

// ═══════════════════════════════════════════════
// FALLBACK (si pas d'API key ou erreur)
// ═══════════════════════════════════════════════

function generateFallbackResult(
  url: string,
  failedCriteria: Array<{ id: string; name: string; category: string; maxPoints: number }>
): MegaGenerationResult {
  
  const solutions: PreGeneratedSolution[] = failedCriteria.map((c, i) => ({
    id: `sol_${String(i + 1).padStart(3, '0')}`,
    category: mapCriteriaCategory(c.category),
    title: `Corriger : ${c.name}`,
    description: `[Simulation] Solution pour le critère "${c.name}". Connectez votre clé API Gemini pour obtenir du code réel.`,
    priority: c.maxPoints >= 4 ? 'critical' : c.maxPoints >= 3 ? 'high' : 'medium',
    estimatedImpact: c.maxPoints,
    fixesCriteria: [c.id],
    mostImpactedEngines: ['perplexity', 'chatgpt'],
    generatedContent: {
      code: `<!-- [Simulation] Code pour "${c.name}" - Connectez votre clé API Gemini -->`,
      language: 'html' as const,
      explanation: `[Simulation] Ce critère impacte votre score de ${c.maxPoints} points.`,
      implementationSteps: [
        'Connectez votre clé API Gemini dans .env.local',
        'Relancez l\'audit pour obtenir les solutions réelles',
      ],
    },
  }));

  return {
    generatedAt: new Date().toISOString(),
    url,
    apiModel: 'fallback-simulation',
    tokenCount: 0,
    solutions,
    fullFAQ: { html: '<!-- Simulation -->', schemaJsonLd: '{}', questionsCount: 0 },
    fullSchemaOrg: { jsonLd: '{}', types: [] },
    fullMetaTags: { title: '', description: '', openGraph: '', canonical: '' },
    fullIntroduction: { html: '<!-- Simulation -->', wordCount: 0 },
    fullComparativeTable: { html: '<!-- Simulation -->', columnsCount: 0, rowsCount: 0 },
    fullArticleOutline: { markdown: '', sections: [], estimatedWordCount: 0 },
    solutionsCount: solutions.length,
    totalEstimatedImpact: solutions.reduce((s, sol) => s + sol.estimatedImpact, 0),
  };
}

function mapCriteriaCategory(cat: string): PreGeneratedSolution['category'] {
  const mapping: Record<string, PreGeneratedSolution['category']> = {
    'structure': 'structure',
    'structured_data': 'schema',
    'authority': 'authority',
    'clarity': 'content',
    'technical': 'technical',
    'geo': 'geo',
  };
  return mapping[cat.toLowerCase()] || 'content';
}

## PHASE 2 — STOCKAGE ET GESTION DES CRÉDITS

### 2.1 — Créer `services/solution-vault.ts` (NOUVEAU FICHIER)

Ce service stocke les solutions pré-générées et gère le verrouillage/déblocage par crédit.

```typescript
// services/solution-vault.ts
//
// VAULT = Coffre-fort des solutions pré-générées.
// Tout est stocké, rien n'est visible sans dépense de crédit.
//
// Architecture :
// - MegaGenerationResult stocké en localStorage (→ Supabase plus tard)
// - Un Set<string> de solutionIds débloqués
// - Quand on débloque : on décompte 1 crédit, on ajoute l'ID au Set
// - Quand on affiche : on vérifie si l'ID est dans le Set

import type { MegaGenerationResult, PreGeneratedSolution } from './mega-generator';

const VAULT_PREFIX = 'citedby_vault_';
const UNLOCKED_PREFIX = 'citedby_unlocked_';

// ═══════════════════════════════════════════════
// TYPES PUBLICS
// ═══════════════════════════════════════════════

export type ContentBlockType = 
  | 'solution'       // Solution individuelle du plan d'action
  | 'fullFAQ'        // FAQ complète
  | 'fullSchemaOrg'  // Schema.org complet
  | 'fullMetaTags'   // Meta tags optimisés
  | 'fullIntro'      // Introduction réécrite
  | 'fullTable'      // Tableau comparatif
  | 'fullOutline';   // Plan d'article

export interface ContentBlock {
  id: string;
  type: ContentBlockType;
  title: string;
  description: string;
  category: string;
  priority: string;
  estimatedImpact: number;
  isUnlocked: boolean;
  
  // Prévisualisation (toujours visible)
  preview: {
    firstLine: string;       // Première ligne du code (teaser)
    language: string;
    linesCount: number;
    wordCount: number;
  };
  
  // Contenu complet (null si verrouillé)
  content: string | null;
  contentLanguage: string | null;
  explanation: string | null;
  implementationSteps: string[] | null;
}

export interface VaultSummary {
  totalBlocks: number;
  unlockedBlocks: number;
  lockedBlocks: number;
  totalImpact: number;
  unlockedImpact: number;
  remainingImpact: number;
  creditCostToUnlockAll: number;
}

// ═══════════════════════════════════════════════
// STOCKAGE
// ═══════════════════════════════════════════════

/**
 * Stocke le résultat du méga-prompt dans le vault.
 * Appelé UNE FOIS après l'audit, automatiquement.
 */
export function storeInVault(auditId: string, result: MegaGenerationResult): void {
  localStorage.setItem(
    `${VAULT_PREFIX}${auditId}`,
    JSON.stringify(result)
  );
  // Initialiser le set des IDs débloqués (vide)
  if (!localStorage.getItem(`${UNLOCKED_PREFIX}${auditId}`)) {
    localStorage.setItem(`${UNLOCKED_PREFIX}${auditId}`, JSON.stringify([]));
  }
}

/**
 * Récupère le résultat stocké pour un audit.
 */
export function getFromVault(auditId: string): MegaGenerationResult | null {
  const data = localStorage.getItem(`${VAULT_PREFIX}${auditId}`);
  return data ? JSON.parse(data) : null;
}

/**
 * Vérifie si un vault existe pour cet audit.
 */
export function hasVault(auditId: string): boolean {
  return localStorage.getItem(`${VAULT_PREFIX}${auditId}`) !== null;
}

// ═══════════════════════════════════════════════
// DÉBLOCAGE PAR CRÉDIT
// ═══════════════════════════════════════════════

/**
 * Récupère la liste des IDs débloqués pour un audit.
 */
function getUnlockedIds(auditId: string): string[] {
  const data = localStorage.getItem(`${UNLOCKED_PREFIX}${auditId}`);
  return data ? JSON.parse(data) : [];
}

/**
 * Débloque un contenu par son ID.
 * NE GÈRE PAS la déduction de crédits — c'est le rôle du composant appelant
 * qui doit vérifier les crédits AVANT d'appeler cette fonction.
 * 
 * @returns true si le déblocage a réussi, false si déjà débloqué
 */
export function unlockContent(auditId: string, contentId: string): boolean {
  const unlocked = getUnlockedIds(auditId);
  if (unlocked.includes(contentId)) return false; // Déjà débloqué
  unlocked.push(contentId);
  localStorage.setItem(`${UNLOCKED_PREFIX}${auditId}`, JSON.stringify(unlocked));
  return true;
}

/**
 * Vérifie si un contenu est débloqué.
 */
export function isUnlocked(auditId: string, contentId: string): boolean {
  return getUnlockedIds(auditId).includes(contentId);
}

/**
 * Débloque TOUT d'un coup (pack complet).
 * @returns nombre de contenus nouvellement débloqués
 */
export function unlockAll(auditId: string): number {
  const vault = getFromVault(auditId);
  if (!vault) return 0;
  
  const allIds = getAllContentIds(vault);
  const currentUnlocked = getUnlockedIds(auditId);
  const newlyUnlocked = allIds.filter(id => !currentUnlocked.includes(id));
  
  localStorage.setItem(
    `${UNLOCKED_PREFIX}${auditId}`,
    JSON.stringify(allIds)
  );
  
  return newlyUnlocked.length;
}

// ═══════════════════════════════════════════════
// CONSTRUCTION DES BLOCS POUR L'AFFICHAGE
// ═══════════════════════════════════════════════

/**
 * Transforme le vault en liste de ContentBlocks prêts à afficher.
 * Chaque bloc sait s'il est verrouillé ou non.
 * Le contenu est null si verrouillé.
 */
export function getContentBlocks(auditId: string): ContentBlock[] {
  const vault = getFromVault(auditId);
  if (!vault) return [];
  
  const unlockedIds = getUnlockedIds(auditId);
  const blocks: ContentBlock[] = [];

  // 1. Solutions individuelles
  for (const sol of vault.solutions) {
    const unlocked = unlockedIds.includes(sol.id);
    blocks.push({
      id: sol.id,
      type: 'solution',
      title: sol.title,
      description: sol.description,
      category: sol.category,
      priority: sol.priority,
      estimatedImpact: sol.estimatedImpact,
      isUnlocked: unlocked,
      preview: {
        firstLine: sol.generatedContent.code.split('\n')[0]?.slice(0, 80) + '...' || '...',
        language: sol.generatedContent.language,
        linesCount: sol.generatedContent.code.split('\n').length,
        wordCount: sol.generatedContent.code.split(/\s+/).length,
      },
      content: unlocked ? sol.generatedContent.code : null,
      contentLanguage: unlocked ? sol.generatedContent.language : null,
      explanation: unlocked ? sol.generatedContent.explanation : null,
      implementationSteps: unlocked ? sol.generatedContent.implementationSteps : null,
    });
  }

  // 2. FAQ complète
  if (vault.fullFAQ.html) {
    const faqId = 'full_faq';
    const unlocked = unlockedIds.includes(faqId);
    blocks.push({
      id: faqId,
      type: 'fullFAQ',

 * Résumé du vault pour affichage.

export function getVaultSummary(auditId: string): VaultSummary {
  const blocks = getContentBlocks(auditId);
  const unlocked = blocks.filter(b => b.isUnlocked);
  const locked = blocks.filter(b => !b.isUnlocked);

  return {
    totalBlocks: blocks.length,
    unlockedBlocks: unlocked.length,
    lockedBlocks: locked.length,
    totalImpact: blocks.reduce((s, b) => s + b.estimatedImpact, 0),
    unlockedImpact: unlocked.reduce((s, b) => s + b.estimatedImpact, 0),
    remainingImpact: locked.reduce((s, b) => s + b.estimatedImpact, 0),
    creditCostToUnlockAll: locked.length,
  };
}

  // 4. Recharger les blocs
  setBlocks(getContentBlocks(auditId));
  setVaultSummary(getVaultSummary(auditId));
};

// Gérer le déblocage de tout
const handleUnlockAll = () => {
  const summary = getVaultSummary(auditId);
  if (!summary) return;
  const packCost = Math.max(1, Math.ceil(summary.lockedBlocks * 0.7));
  if (creditsRemaining < packCost) return;
  
  // Déduire les crédits du pack
  // ⚠️ ADAPTER
  
  unlockAll(auditId);
  setBlocks(getContentBlocks(auditId));
  setVaultSummary(getVaultSummary(auditId));
};

// Dans le JSX, REMPLACER la liste de solutions par :
// (garder le score projeté et le header au-dessus)

{vaultSummary && (
  <UnlockAllBanner
    summary={vaultSummary}
    onUnlockAll={handleUnlockAll}
    creditsRemaining={creditsRemaining}
  />
)}

<div className="mt-6 space-y-4">
  {blocks.map(block => (
    <LockedContentCard
      key={block.id}
      block={block}
      onUnlock={handleUnlock}
      creditsRemaining={creditsRemaining}
    />
  ))}
</div>

{blocks.length === 0 && (
  <div className="text-center py-12 text-slate-500">
    <p>Lancez un audit pour générer les solutions.</p>
  </div>
)}
```

### 5.1 — Créer `config/credit-costs.ts` (NOUVEAU FICHIER)

```typescript
// config/credit-costs.ts
//
// Définition du coût en crédits de chaque action du SaaS.
// Un seul fichier de référence pour toute la tarification.

export const CREDIT_COSTS = {
  // ═══ AUDIT ═══
  runAudit: 1,                    // Lancer un audit AEO (inclut le méga-prompt)
  
  // ═══ DÉBLOCAGE DE SOLUTIONS (plan d'action) ═══
  unlockSolution: 1,              // Débloquer 1 solution individuelle
  unlockFullFAQ: 2,               // Débloquer la FAQ complète (plus de valeur)
  unlockFullSchema: 1,            // Débloquer le Schema.org complet
  unlockFullMeta: 1,              // Débloquer les meta tags
  unlockFullIntro: 1,             // Débloquer l'introduction
  unlockFullTable: 1,             // Débloquer le tableau comparatif
  unlockFullOutline: 1,           // Débloquer le plan d'article
  
  // ═══ PACK COMPLET ═══
  // Le pack = 70% du coût total (30% de réduction)
  unlockAllDiscount: 0.7,
  
  // ═══ AEO STUDIO (outils individuels) ═══
  generateFAQ: 1,                 // Générer une FAQ depuis le studio
  generateSchema: 1,              // Générer un Schema.org
  generateArticle: 2,             // Générer un article complet
  generateProductSheet: 1,        // Générer une fiche produit
  
  // ═══ CITATIONS V3 ═══
  runCitationCheck: 2,            // Lancer une vérification de citations
  // (1 vérification = toutes les requêtes × 3 moteurs)
  
  // ═══ RE-SCAN ═══
  rescan: 1,                      // Re-scanner une URL
} as const;

/**
 * Calcule le coût du pack "tout débloquer".
 */
export function calculatePackCost(lockedCount: number): number {
  return Math.max(1, Math.ceil(lockedCount * CREDIT_COSTS.unlockAllDiscount));
}

/**
 * Retourne le coût d'un déblocage par type de contenu.
 */
export function getUnlockCost(contentType: string): number {
  switch (contentType) {
    case 'fullFAQ': return CREDIT_COSTS.unlockFullFAQ;
    case 'fullSchemaOrg': return CREDIT_COSTS.unlockFullSchema;
    case 'fullMetaTags': return CREDIT_COSTS.unlockFullMeta;
    case 'fullIntro': return CREDIT_COSTS.unlockFullIntro;
    case 'fullTable': return CREDIT_COSTS.unlockFullTable;
    case 'fullOutline': return CREDIT_COSTS.unlockFullOutline;
    default: return CREDIT_COSTS.unlockSolution;
  }
}
```
